    134. Gas Station

    Link: https://leetcode.com/problems/gas-station/

    Solution: greedy

    
# Time Complexity: O(n)
# Space Complexity: O(1)

class Solution:
    def canCompleteCircuit(self, gas: List[int], cost: List[int]) -> int:
        
        if sum(gas) < sum(cost):
            return -1
        start = 0
        total = 0
        
        for i in range(len(gas)):
            total += gas[i] - cost[i]
            
            if total < 0:
                start = i + 1
                total = 0
        return start
